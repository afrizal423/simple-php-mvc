# Tugas mata kuliah Framework
```
├── app                           # Direktori file MVC utama
│   ├── controllers               # Controllers Direktori
│   │   └── example.php           # Contoh Controller dengan fungsinya 
│   ├── models                    # Models Direktori
│   │   └── example.php           # Contoh Model dengan dengan fungsinya 
│   └── views                     # Views Direktori
├── core                          # Direktori logic mvc
│   ├── app.php                   # File framework utama
│   ├── classes                   # Direktori Classes untuk login controller dan model
│   │   ├── controller.php        
│   │   └── model.php             
│   ├── config                    # Direktori Konfigurasi
│   │   ├── database.php          
│   │   └── session.php           
│   └── helpers                   # Autoloaded helpers directory
│       └── examplehelper.php     
├── index.php                     # Endpoint url
├── public                        # Direktori untuk semua asset publik, file javascript, stylesheet, dan plugin vendor
│   ├── javascripts               
│   ├── stylesheets               
│   └── vendor                    
└── .htaccess                     # htaccess menulis ulang semua requests to MVC endpoint /index.php
```
